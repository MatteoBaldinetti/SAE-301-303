Table methodes {
  id_methode int [not null, pk, increment]
  nom_methode varchar
  nom_famille varchar
  sous_famille varchar
  autre_caracteristique varchar
}

table culture {
  id_culture int [pk, not null, increment]
  cultures varchar
  filiere varchar
  terre_abri varchar
}

table cible {
  id_cible int [pk, not null, increment]
  cible_principale varchar
  groupe_cible varchar
}

table projet {
  id_projet int [pk, not null, increment]
  nom_projet varchar
  periode varchar
  code_postal int
}

table traitement {
  id_traitement int [pk, not null, increment]
  type varchar
  partie varchar
  mode_action varchar
}

table application {
  id_application int [pk, not null, increment]
  stade varchar
  nombre int
  dose varchar
  rdd_oad varchar
}

table experimentation {
  id_experimentation int [pk, not null, increment]
  nom_system varchar
  site_experimental varchar
  lien varchar
}

ref: methodes.id_methode <> projet.id_projet
ref: application.id_application <> methodes.id_methode
ref: methodes.id_methode <> cible.id_cible
ref: experimentation.id_experimentation <> projet.id_projet
ref: culture.id_culture <> projet.id_projet
ref: traitement.id_traitement <> methodes.id_methode
