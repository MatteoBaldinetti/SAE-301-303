Table methodes {
  id_methode int [not null, pk, increment]
  nom_methode varchar
  nom_famille varchar
  sous_famille varchar
  autre_caracteristique varchar
}

table culture {
  id_culture int [pk, not null, increment]
  cultures varchar
  filiere varchar
  terre_abri varchar
}

table cible {
  id_cible int [pk, not null, increment]
  cible_principale varchar
  groupe_cible varchar
}

table projet {
  id_projet int [pk, not null, increment]
  nom_projet varchar
  nom_system varchar
  periode timestamp
  site_experimental varchar
  code_postal int
}

table traitement {
  id_traitement int [pk, not null, increment]
  type varchar
  partie varchar
  mode_action varchar
}

table application {
  id_application int [pk, not null, increment]
  stade varchar
  nombre int
  dose varchar
  rdd_aod varchar
}

ref: methodes.id_methode <> projet.id_projet
ref: traitement.id_traitement <> cible.id_cible
ref: traitement.id_traitement <> culture.id_culture
ref: methodes.id_methode <> traitement.id_traitement
ref: traitement.id_traitement <> application.id_application
